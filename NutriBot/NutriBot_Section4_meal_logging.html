def query_usda(food_item):
    API_KEY = "YOUR_USDA_API_KEY"
    url = f"https://api.nal.usda.gov/fdc/v1/foods/search?query={food_item}&api_key={API_KEY}"
    res = requests.get(url).json()
    top_result = res["foods"][0]
    return {
        "name": top_result["description"],
        "calories": top_result.get("foodNutrients", [{}])[0].get("value", 0),
        "protein": top_result.get("foodNutrients", [{}])[1].get("value", 0),
        # expand with sodium, fat, etc.
    }

  • Version control: Food mapping dictionary is versioned to ensure consistency across sessions
